    }

        Node next;

        Node(T value, Node next) {

            this.next = next;
        }
    }

    public void push(T value) {
        top = new Node(value, top);

    }

        if (isEmpty()) throw new RuntimeException("Stack underflow");
        T val = top.value;
        top = top.next;

    }

        return top.value;
    }

}
